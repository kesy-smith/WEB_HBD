Copiez ces commandes dans la console (F12) pour les tester:

Génération de Souhaits

// Créer un générateur
const gen = new WishesGenerator();

// Générer un souhait nostalgique
gen.generateWish('nostalgique', 'Alice', 25);

// Générer avec fortune
gen.generateCompleteSalutation('explosive', 'Bob', 30, 'professional');

// Valider l'entrée
gen.validateInput('mystique', 'Charlie', 35, 'romantic');

Contrôle des Particules

// Intensifier les particules
particleSystem.intensify();

// Exploser les particules
particleSystem.explodeCenter();

// Changer les couleurs
particleSystem.setEmotionalColors('galactique');

// Détruire le système
particleSystem.destroy();

Portail AR

// Ouvrir le portail
arSimulator.open("Joyeux anniversaire!");

// Créer des confettis
arSimulator.createConfetti();

// Afficher un hologramme
arSimulator.createWishHologram("Mon souhait");

// Jouer un son
arSimulator.playARSound();

Contrôle de l'App

// Générer un souhait
app.generateWish();

// Afficher un toast
app.showToast('Message personnalisé');

// Copier le souhait
app.copyWish();

// Voir l'état
console.log(app.state);
